      function solveLinearSystem(A, B) {

        const n = 8;

        for (let i = 0; i < n; i++) {

          let maxRow = i;

          for (let k = i + 1; k < n; k++) {

            if (Math.abs(A[k][i]) > Math.abs(A[maxRow][i])) maxRow = k;

          }

          const tempRow = A[i]; A[i] = A[maxRow]; A[maxRow] = tempRow;

          const tempVal = B[i]; B[i] = B[maxRow]; B[maxRow] = tempVal;

          

          for (let k = i + 1; k < n; k++) {

            const c = -A[k][i] / A[i][i];

            for (let j = i; j < n; j++) {

              if (i === j) A[k][j] = 0;

              else A[k][j] += c * A[i][j];

            }

            B[k] += c * B[i];

          }

        }

        const x = new Array(n).fill(0);

        for (let i = n - 1; i >= 0; i--) {

          let sum = 0;

          for (let j = i + 1; j < n; j++) sum += A[i][j] * x[j];

          x[i] = (B[i] - sum) / A[i][i];

        }

        return x;

      }


      function getPerspectiveTransform(srcPts, dstPts) {

        const A = [];

        const B = [];

        for (let i = 0; i < 4; i++) {

          const sx = srcPts[i].x;

          const sy = srcPts[i].y;

          const dx = dstPts[i].x;

          const dy = dstPts[i].y;

          

          A.push([dx, dy, 1, 0, 0, 0, -dx * sx, -dy * sx]);

          B.push(sx);

          

          A.push([0, 0, 0, dx, dy, 1, -dx * sy, -dy * sy]);

          B.push(sy);

        }

        return solveLinearSystem(A, B);

      }

      function applyMonochromeBwFilter(base64Str, threshold = 165, contrastRange = 35, enabled = true) {

        if (!enabled) return Promise.resolve(base64Str);


        return new Promise((resolve, reject) => {

          const img = new Image();

          img.crossOrigin = "anonymous";

          img.onload = () => {

            try {

              const canvas = document.createElement('canvas');

              canvas.width = img.width;

              canvas.height = img.height;

              const ctx = canvas.getContext('2d');

              ctx.drawImage(img, 0, 0);

              

              const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height);

              const data = imgData.data;

              

              const transitionWidth = Math.max(5, contrastRange); 

              const lowerBound = Math.max(0, threshold - transitionWidth);

              const upperBound = Math.min(255, threshold + transitionWidth);

              

              for (let i = 0; i < data.length; i += 4) {

                const r = data[i];

                const g = data[i+1];

                const b = data[i+2];

                const gray = 0.299 * r + 0.587 * g + 0.114 * b;

                

                let finalVal;

                if (gray > upperBound) {

                  finalVal = 255; 

                } else if (gray < lowerBound) {

                  finalVal = Math.max(0, gray * 0.4); 

                } else {

                  const ratio = (gray - lowerBound) / (upperBound - lowerBound);

                  const darkTarget = Math.max(0, lowerBound * 0.4);

                  finalVal = darkTarget + ratio * (255 - darkTarget);

                }

                

                data[i] = finalVal;     

                data[i+1] = finalVal;   

                data[i+2] = finalVal;   

              }

              

              ctx.putImageData(imgData, 0, 0);

              resolve(canvas.toDataURL('image/png'));

            } catch (err) {

              reject(err);

            }

          };

          img.onerror = (err) => reject(err);

          img.src = base64Str;

        });

      }

      async function callGeminiWithRetry(url, payload, retries = 5, delay = 1000) {

        let lastError = null;

        for (let i = 0; i < retries; i++) {

          try {

            const response = await fetch(url, {

              method: 'POST',

              headers: { 'Content-Type': 'application/json' },

              body: JSON.stringify(payload)

            });

            

            if (response.ok) {

              return await response.json();

            }


            let errorMsg = `HTTP Error ${response.status}`;

            try {

              const errJson = await response.json();

              if (errJson.error?.message) {

                errorMsg = errJson.error.message;

              }

            } catch (e) {}


            if (response.status === 400 || response.status === 401 || response.status === 403) {

              throw new Error(errorMsg);

            }

            lastError = new Error(errorMsg);

          } catch (err) {

            lastError = err;

            const msg = err.message || "";

            if (msg.includes("API key") || msg.includes("not valid") || msg.includes("400") || msg.includes("403")) {

              throw err;

            }

          }

          if (i < retries - 1) {

            await new Promise(res => setTimeout(res, delay));

            delay *= 2; 

          }

        }

        throw lastError || new Error("實在頂唔住，API 連接失敗！請確認網絡或設置 Key！");

      }

      function downloadSingleCleanedImage(base64Data, pageName = 'Cleaned_Worksheet') {

        if (!base64Data) return;

        try {

          if (base64Data.startsWith('data:')) {

            const parts = base64Data.split(',');

            const mime = parts[0].match(/:(.*?);/)[1];

            const rawBase64 = parts[1];

            const binaryStr = atob(rawBase64);

            const len = binaryStr.length;

            const u8arr = new Uint8Array(len);

            for (let i = 0; i < len; i++) {

              u8arr[i] = binaryStr.charCodeAt(i);

            }

            const blob = new Blob([u8arr], { type: mime });

            const blobUrl = URL.createObjectURL(blob);

            const link = document.createElement('a');

            link.href = blobUrl;

            link.download = `${pageName}.png`;

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

            setTimeout(() => URL.revokeObjectURL(blobUrl), 200);

          }

        } catch (error) {

          openImageInNewTab(base64Data);

        }

      }
